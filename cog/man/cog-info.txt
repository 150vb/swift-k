NAME
    cog-info - Displays all the relevant Grid information of the
    client machine

SYNOPSIS
    cog-info [-all] [-user] [-os] [-java] [-ant] [-provider] [-env] [-help] 

OPTIONS
  cog-info <options>
    [-all]
      Display all the client-side information

    [-os]
      Display the operating system on the client machine

    [-user]
      Display the username

    [-java]
      Display all the java-specific information

    [-ant]
      Display all the Apache Ant specific information

    [(-providers | -p)]
      Display the list of supported providers

    [-env]
      Prints all the environment variables

    [(-help | -h)]
      Display usage
